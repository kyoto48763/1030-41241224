動畫展示

Lottie.js 官網

這是一個簡單的動畫展示，使用 Lottie.js 加載 JSON 格式動畫。

範例程式碼


            <!-- Lottie.js -->
            <script src="https://cdnjs.cloudflare.com/ajax/libs/lottie-web/5.7.13/lottie.min.js"></script>
            
            <!-- Lottie 動畫初始化 -->
            <script>
                lottie.loadAnimation({
                    container: document.getElementById('lottie-animation'), // 容器
                    renderer: 'svg', // 渲染方式
                    loop: true, // 是否循環播放
                    autoplay: true, // 自動播放
                    path: 'YOUR_JSON_FILE_URL.json' // JSON 動畫檔案路徑
                });
            </script>
